import React from 'react'; import { useState, useEffect, useReducer} from 'react';

import 'bootstrap/dist/css/bootstrap.min.css'; import Container from 'react-bootstrap/Container'; import Row from 'react-bootstrap/Row'; import Col from 'react-bootstrap/Col'; import Form from 'react-bootstrap/Form'; import Button from 'react-bootstrap/Button'; import Table from 'react-bootstrap/Table';

function reducer(state,event) { switch (event.type) { case "FETCH": return { …state, status: "loading" }; case "RESOLVE": return { …state, status: "success", data: event.data }; case "REJECT": return { …state, status: "failure", error: event.error }; case "CANCEL": return { …state, status: "idle", }; default: return state; } }

const initialState = { status: "idle", data: null, error: null };

const List = props => { let list; if(props.data.data !== undefined) { list = props.data.data.map(elem => { let todoTitle = !elem.isCompleted ? elem.data : <strike><b>{elem.data}</b></strike>; return ( <tr key={'key-'+elem.id}> <td>{todoTitle} <Button onClick={() > props.onClick(elem._id, 'complete' ) } variant"info" size="sm" className="float-right ml-2"> Complete </Button> <Button onClick={() > props.onClick(elem._id, 'delete' ) } variant"danger" size="sm" className="float-right"> Delete </Button> </td> </tr>); }); } else { list = null; } return list; };

function fetcher() { const [state, dispatch] = useReducer(reducer, initialState); const [error, data, status] = state;

useEffect(() > { if(state.status == "loading") { let canceled = false;

fetch() .then( data => { if(canceled) return; dispatch({type: "RESOLVE", data}); }) .catch(error => { if(canceled) return; dispatch({type: "REJECT", error}); } );

return () => { canceled = true; }; } }, [state.status]);

return(); }

function App() {

let [serverData, setServerData] = useState([{data: []}]); let [timeStamp, setTimeStamp] = useState(''); let [query, setQuery] = useState('');

useEffect( () => { fetch('http://localhost:5000/todo', {method: 'GET'}) .then( res => res.json()) .then( dat => setServerData(dat)); },[timeStamp]);

const handleClick = (id, action) => { switch (action) { case 'delete': fetch('http://localhost:5000/todo', { method: 'DELETE', headers: { Accept: 'application/json', 'Content-Type': 'application/json' }, body: JSON.stringify({id: id}) }) .then(res => res.json()) .then(timeStamp=> setTimeStamp(timeStamp.timeStamp)); break; case 'complete': fetch('http://localhost:5000/todo', { method: 'PUT', headers: { Accept: 'application/json', 'Content-Type': 'application/json' }, body: JSON.stringify({id: id}) }) .then(res => res.json()) .then(timeStamp=> setTimeStamp(timeStamp.timeStamp)); break; } };

const handleSubmit = evt => { evt.preventDefault(); fetch('http://localhost:5000/todo', { method: 'POST', headers: { Accept: 'application/json', 'Content-Type': 'application/json' }, body: JSON.stringify({data: query}) }) .then(setQuery('')) .then(res => res.json()) .then(dta=> setTimeStamp(dta.timeStamp)); };

return ( <Container> <Row > <Col xs={12} className="mt-5"> <h1 >To do:</h1> <Form className="mt-2" onSubmit={handleSubmit} > <Form.Control type="text" required placeholder="What needs to be done?" onChange={evt => setQuery(evt.target.value)} value={query} /> <Button className="mt-2 float-right" variant="primary" type="submit"> Add Todo </Button> </Form> </Col> <Col> <Table bordered hover className="mt-5" > <tbody> <List data={serverData} onClick={handleClick}/> </tbody> </Table> </Col> </Row> </Container> ); }

export default App;
